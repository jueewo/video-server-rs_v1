---
import MainLayout from "../../../layouts/main.astro";
import ElementRenderer from "../../../components/page-renderer/ElementRenderer.astro";
import { getCollection } from "astro:content";
import Config from "../../../website.config.cjs";

export const prerender = true;

const { lang } = Astro.params;

export function getStaticPaths() {
    return {{langstaticarray}};
}

let page_metatags = {
    title: "{{metatag_title}}",
    description: "{{metatag_description}}",
    keywords: "{{metatag_keywords}}",
    author: "{{metatag_author}}",
};

const title = "{{title}}";
const collection_name = "{{collection_name}}";
const default_lang = Config.defaultLanguage.locale;

// Each locale has one page.json file — Astro content id is "{locale}/page"
const all_entries = await getCollection(collection_name);
const find_entry = (l: string) => all_entries.find((e) => e.id === l + "/page");

let lang_entry = find_entry(lang ?? default_lang);
let showing_default_language = false;

if (!lang_entry || !lang_entry.data?.elements?.length) {
    lang_entry = find_entry(default_lang);
    showing_default_language = !!lang_entry;
}

let page_elements: any[] = ((lang_entry?.data?.elements as any[]) ?? []).filter(
    (el: any) => !el.draft,
);

// Extract Page-Metatags element if present
const metatags_idx = page_elements.findIndex(
    (el: any) => el.element === "Page-Metatags",
);
if (metatags_idx > -1) {
    const [metatags_el] = page_elements.splice(metatags_idx, 1);
    // Read from content (canonical); fall back to props for legacy data
    const src = metatags_el?.content ?? metatags_el?.props ?? {};
    if (Object.keys(src).length) {
        page_metatags = {
            title:       src.title       ?? page_metatags.title,
            description: src.description ?? "",
            keywords:    src.keywords    ?? "",
            author:      src.author      ?? "",
        };
    }
}

const info = {
    draft: false,
    weight: 0,
    element: "TitleAlertBanner",
    slot: null,
    wrapper: null,
    content: {
        title:
            "This page is not available in your selected language => showing " +
            Config.defaultLanguage.language,
        desc: null,
    },
    props: { id: "introhero", h2: true },
};
---

<MainLayout title={title} page_metatags={page_metatags}>
    {showing_default_language && (
        <div class="">
            <ElementRenderer element={info} index={1} />
        </div>
    )}
    {page_elements.map((element: any, index: number) => (
        <ElementRenderer element={element} index={index} />
    ))}
</MainLayout>
